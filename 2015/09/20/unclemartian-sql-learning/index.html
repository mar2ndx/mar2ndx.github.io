Date: September 20, 2015  0:00:00 Category: unknown

Raw data: 

drop table Frequents;
drop table Sells;
drop table Likes;
drop table Drinkers;
drop table Bars;
drop table Beers;

create table Beers (
	name varchar(50),
	manf varchar(100),
	primary key (name)
);

create table Bars (
	name varchar(50),
	addr varchar(100),
	primary key (name)
);

create table Drinkers (
	name varchar(50),
	addr varchar(100),
	phone int,
	primary key (name)
);

create table Likes (
	drinker varchar(50),
	beer varchar(50),
	primary key (drinker, beer)
);

create table Sells (
	bar varchar(50),
	beer varchar(50),
	price float
);

create table Frequents (
	drinker varchar(50),
	bar varchar(50)
);
